\vspace{\baselineskip}

\begin{center}
\begin{tabular}{l|c}
\textbf{Description} & \textbf{Amount} \\
\hline

\emph{Book}: Introducing Character Animation with Blender & \pounds 18.99 \\
\textbf{Total} & \pounds 48.98
\end{tabular}
\end{center}
